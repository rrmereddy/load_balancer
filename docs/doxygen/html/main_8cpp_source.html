32/**
33 * @brief Generates a request biased toward blocked or allowed traffic.
34 * @param ipBlocker Configured allowlist checker.
35 * @param blockedTrafficRate Target probability of generating blocked traffic.
36 * @return Generated request matching target distribution when possible.
37 */
38static Request makeBiasedRequest(const IpBlocker& ipBlocker, double blockedTrafficRate) {
39
40 static std::mt19937 engine(std::random_device{}());
41 double clampedRate = blockedTrafficRate;
42 if (clampedRate < 0.0) {
43 clampedRate = 0.0;
44 } else if (clampedRate > 1.0) {
45 clampedRate = 1.0;
46 }
47
48 const bool shouldBeBlocked = std::bernoulli_distribution(clampedRate)(engine);
49 Request fallback = makeRandomRequest();
50 bool fallbackBlocked = !ipBlocker.isAllowed(fallback.ipIn);
51 if (fallbackBlocked == shouldBeBlocked) {
52 return fallback;
53 }
54
55 // Keep generation simple while biasing toward the requested blocked/allowed ratio.
56 for (int i = 0; i < 32; ++i) {
57 Request candidate = makeRandomRequest();
58 const bool candidateBlocked = !ipBlocker.isAllowed(candidate.ipIn);
59 if (candidateBlocked == shouldBeBlocked) {
60 return candidate;
61 }
62 }
63 return fallback;
64}

66/**
67 * @brief Simulates one cycle of incoming traffic generation and routing.
68 * @param switchRouter Switch used to route accepted requests.
69 * @param ipBlocker Allowlist checker for source IPs.
70 * @param blockedTrafficRate Target blocked-traffic ratio for generation.
71 * @param currentClock Current simulation cycle for logging.
72 * @param logger Logger used for traffic event output.
73 * @return Accepted/rejected counts per job type for this cycle.
74 */
75static IncomingTrafficResult simulateIncomingRequests(
76 Switch& switchRouter,
77 const IpBlocker& ipBlocker,
78 double blockedTrafficRate,
79 int currentClock,
80 Logger& logger) {
81 static std::mt19937 engine(std::random_device{}());
82 std::bernoulli_distribution burstChance(0.70);
83 std::uniform_int_distribution<int> newRequestDist(1, 10);
84 IncomingTrafficResult result = {0, 0, 0, 0};
85
86 if (!burstChance(engine)) {
87 return result;
88 }
89
90 const int newRequests = newRequestDist(engine);
91 std::string firstRejectedIp;
92 for (int i = 0; i < newRequests; ++i) {
93 const Request request = makeBiasedRequest(ipBlocker, blockedTrafficRate);
94 if (!ipBlocker.isAllowed(request.ipIn)) {
95 if (request.jobType == JobType::Streaming) {
96 ++result.rejectedStreamingRequests;
97 } else {
98 ++result.rejectedProcessingRequests;
99 }
100 if (firstRejectedIp.empty()) {
101 firstRejectedIp = request.ipIn;
102 }
103 continue;
104 }
105 switchRouter.routeRequest(request);
106 if (request.jobType == JobType::Streaming) {
107 ++result.acceptedStreamingRequests;
108 } else {
109 ++result.acceptedProcessingRequests;
110 }
111 }
112
113 const int totalAccepted = result.acceptedProcessingRequests + result.acceptedStreamingRequests;
114 const int totalRejected = result.rejectedProcessingRequests + result.rejectedStreamingRequests;
115
116 if (totalAccepted > 0) {
117 logger.log("Clock " + std::to_string(currentClock) +
118 ": Incoming traffic -> accepted " + std::to_string(totalAccepted) +
119 " request(s), processing=" + std::to_string(result.acceptedProcessingRequests) +
120 ", streaming=" + std::to_string(result.acceptedStreamingRequests));
121 }
122
123 if (totalRejected > 0) {
124 logger.log("Clock " + std::to_string(currentClock) +
125 ": IP range blocker rejected " + std::to_string(totalRejected) +
126 " request(s), processing=" + std::to_string(result.rejectedProcessingRequests) +
127 ", streaming=" + std::to_string(result.rejectedStreamingRequests) +
128 " request(s)");
129 if (!firstRejectedIp.empty()) {
130 logger.log("Clock " + std::to_string(currentClock) +
131 ": Sample rejected source IP: " + firstRejectedIp);
132 }
133 }
134 return result;
135}

161
162 IpBlocker ipBlocker;
163 for (const auto& range : config.allowedIpRanges) {

165 logger.log("Invalid allowed IP range ignored: " + range.start + " - " + range.end);

169
170 const int initialRequestTarget = initialServers * 100;
171 int initialAcceptedProcessingRequests = 0;
172 int initialAcceptedStreamingRequests = 0;
173 int initialRejectedProcessingRequests = 0;
174 int initialRejectedStreamingRequests = 0;
175
176 for (int i = 0; i < initialRequestTarget; ++i) {
177 const Request request = makeBiasedRequest(ipBlocker, config.blockedTrafficSimulationRate);
178 if (!ipBlocker.isAllowed(request.ipIn)) {
179 if (request.jobType == JobType::Streaming) {
180 ++initialRejectedStreamingRequests;
181 } else {
182 ++initialRejectedProcessingRequests;
183 }
184 continue;
185 }
186 switchRouter.routeRequest(request);
187
188 if (request.jobType == JobType::Streaming) {
189 ++initialAcceptedStreamingRequests;
190 } else {
191 ++initialAcceptedProcessingRequests;
192 }
193 }
194
195 const int initialAcceptedRequests = initialAcceptedProcessingRequests + initialAcceptedStreamingRequests;
196 const int initialRejectedRequests = initialRejectedProcessingRequests + initialRejectedStreamingRequests;
197 if (initialRejectedRequests > 0) {
198 logger.log("Initial request load -> accepted " + std::to_string(initialAcceptedRequests) +
199 " (processing=" + std::to_string(initialAcceptedProcessingRequests) +
200 ", streaming=" + std::to_string(initialAcceptedStreamingRequests) + ")" +
201 ", rejected " + std::to_string(initialRejectedRequests) +
202 " (processing=" + std::to_string(initialRejectedProcessingRequests) +
203 ", streaming=" + std::to_string(initialRejectedStreamingRequests) + ")");
204 }
205
206 LoadBalancerMetrics processingMetrics = initializeMetrics(processingBalancer);
207 LoadBalancerMetrics streamingMetrics = initializeMetrics(streamingBalancer);
208
209 logSimulationStartSnapshot(

214 runCycles,

241 processingBalancer.evaluateScaling(config.minQueuePerServer, config.maxQueuePerServer, logger);
242 streamingBalancer.evaluateScaling(config.minQueuePerServer, config.maxQueuePerServer, logger);

244

258 streamingMetrics.totalRejectedRequests += initialRejectedStreamingRequests;
259
260 logSimulationEndSummary(

262 "Processing",

264 processingMetrics,

267 logSimulationEndSummary(
268 logger,
269 "Streaming",
270 streamingBalancer,
271 streamingMetrics,
272 streamingInitialServers,
273 cyclesRun);
274
275 const int totalServers = processingBalancer.getServerCount() + streamingBalancer.getServerCount();
276 const int totalRequests = processingBalancer.getTotalRequestsCount() + streamingBalancer.getTotalRequestsCount();
277 const int totalHandled =
278 processingBalancer.getTotalRequestsHandledCount() + streamingBalancer.getTotalRequestsHandledCount();
279 const int totalRemaining =
280 processingBalancer.getTotalRequestsRemainingCount() + streamingBalancer.getTotalRequestsRemainingCount();
281
282 std::cout << "Simulation complete. Servers: " << totalServers
283 << ", Requests: " << totalRequests
284 << ", Cycles run: " << cyclesRun
285 << ", Requests handled: " << totalHandled

288

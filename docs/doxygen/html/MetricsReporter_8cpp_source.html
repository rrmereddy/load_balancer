25 * @param request Request to format.
26 * @return Formatted request line.

28std::string formatRequestForLog(const Request& request) {
29 std::ostringstream line;
30 line << request.ipIn << " -> " << request.ipOut
31 << " | type=" << jobTypeToString(request.jobType)
32 << " | time=" << request.timeCycles;

48/**
49 * @brief Logs a snapshot of the balancer queue.
50 * @param logger Logger used for output.
51 * @param balancer Balancer whose queue is sampled.
52 * @param title Prefix title for this snapshot.
53 * @param maxItems Maximum queued requests to print.
54 */
55void logQueueSnapshot(Logger& logger, const LoadBalancer& balancer, const std::string& title, std::size_t maxItems) {
56 logger.log(title + " queue size: " + std::to_string(balancer.getQueueSize()));
57 const std::vector<Request> snapshot = balancer.getQueueSnapshot(maxItems);
58
59 if (snapshot.empty()) {
60 logger.log(title + " queue sample: [empty]");
61 return;
62 }
63
64 logger.log(title + " queue sample (first " + std::to_string(snapshot.size()) + " request(s)):");
65 for (std::size_t i = 0; i < snapshot.size(); ++i) {
66 logger.log(" [" + std::to_string(i + 1) + "] " + formatRequestForLog(snapshot[i]));
67 }
68}

70/**
71 * @brief Logs a snapshot of server utilization and sampled assignments.
72 * @param logger Logger used for output.
73 * @param balancer Balancer whose servers are sampled.
74 * @param title Prefix title for this snapshot.
75 * @param maxItems Maximum servers to print.
76 */
77void logServerSnapshot(Logger& logger, const LoadBalancer& balancer, const std::string& title, std::size_t maxItems) {
78 const std::vector<ServerSnapshot> snapshots = balancer.getServerSnapshots();
79 logger.log(title + " server snapshot: active=" + std::to_string(balancer.getActiveServerCount()) +
80 ", inactive=" + std::to_string(balancer.getIdleServerCount()) +
81 ", total=" + std::to_string(balancer.getServerCount()));
82
83 if (snapshots.empty()) {
84 logger.log(" [no servers available]");
85 return;
86 }
87
88 const std::size_t sampleCount = std::min(maxItems, snapshots.size());
89 for (std::size_t i = 0; i < sampleCount; ++i) {
90 const ServerSnapshot& server = snapshots[i];
91 if (!server.busy) {
92 logger.log(" Server " + std::to_string(server.id) + ": idle");
93 continue;
94 }
95 logger.log(" Server " + std::to_string(server.id) + ": busy -> " + formatRequestForLog(server.request));
96 }
97}

101/**
102 * @brief Initializes metrics from the balancer's starting state.
103 * @param balancer Balancer to initialize metrics for.
104 * @return Initialized metrics structure.
105 */
106LoadBalancerMetrics initializeMetrics(const LoadBalancer& balancer) {
107 const std::size_t startingQueueSize = balancer.getQueueSize();
108 LoadBalancerMetrics metrics;
109 metrics.startingQueueSize = startingQueueSize;
110 metrics.peakQueueSize = startingQueueSize;
111 metrics.queueSizeAccumulator = static_cast<long long>(startingQueueSize);
112 metrics.queueSamples = 1;
113 metrics.totalIncomingRequests = 0;
114 metrics.totalRejectedRequests = 0;
115 metrics.minServersObserved = balancer.getServerCount();
116 metrics.maxServersObserved = balancer.getServerCount();
117 return metrics;
118}

120/**
121 * @brief Updates metrics after one simulation cycle.
122 * @param metrics Metrics object to mutate.
123 * @param balancer Balancer providing state values.
124 * @param acceptedRequests Accepted requests for this cycle.
125 * @param rejectedRequests Rejected requests for this cycle.
126 */
127void updateMetrics(
128 LoadBalancerMetrics& metrics,
129 const LoadBalancer& balancer,
130 int acceptedRequests,
131 int rejectedRequests) {
132 metrics.totalIncomingRequests += acceptedRequests;
133 metrics.totalRejectedRequests += rejectedRequests;
134 const std::size_t queueSize = balancer.getQueueSize();
135 metrics.peakQueueSize = std::max(metrics.peakQueueSize, queueSize);
136 metrics.queueSizeAccumulator += static_cast<long long>(queueSize);
137 ++metrics.queueSamples;
138 metrics.minServersObserved = std::min(metrics.minServersObserved, balancer.getServerCount());
139 metrics.maxServersObserved = std::max(metrics.maxServersObserved, balancer.getServerCount());
140}

175 * @brief Logs the end-of-run summary for a balancer.
176 * @param logger Logger used for output.
177 * @param label Balancer label for log scoping.
178 * @param balancer Balancer being reported.
179 * @param metrics Aggregated run metrics.

181 * @param cyclesRun Number of cycles completed.

183void logSimulationEndSummary(
184 Logger& logger,
185 const std::string& label,
186 const LoadBalancer& balancer,
187 const LoadBalancerMetrics& metrics,
188 int initialServers,
189 int cyclesRun) {
190 const std::string prefix = "[" + label + "] ";
191 const std::size_t endingQueueSize = balancer.getQueueSize();
192 const int totalRequests = balancer.getTotalRequestsCount();
193 const int requestsHandled = balancer.getTotalRequestsHandledCount();
194 const int requestsRemaining = balancer.getTotalRequestsRemainingCount();
195
196 const double averageQueueSize = metrics.queueSamples > 0
197 ? static_cast<double>(metrics.queueSizeAccumulator) /
198 static_cast<double>(metrics.queueSamples)
199 : 0.0;
200 const double throughputPerCycle = cyclesRun > 0
201 ? static_cast<double>(requestsHandled) / static_cast<double>(cyclesRun)
202 : 0.0;
203 const double completionRatio = totalRequests > 0
204 ? (100.0 * static_cast<double>(requestsHandled)) /
205 static_cast<double>(totalRequests)
206 : 0.0;
207 const int totalGeneratedDuringRun = metrics.totalIncomingRequests + metrics.totalRejectedRequests;
208 const int totalGeneratedIncludingInitial = totalRequests + metrics.totalRejectedRequests;
209 const double rejectionRatio = totalGeneratedIncludingInitial > 0
210 ? (100.0 * static_cast<double>(metrics.totalRejectedRequests)) /
211 static_cast<double>(totalGeneratedIncludingInitial)
212 : 0.0;
213
214 logger.log(prefix + "=== Simulation End Summary ===");
215 logger.log(prefix + "");
216 logger.log(prefix + "Run");
217 logger.log(prefix + " Cycles run: " + std::to_string(cyclesRun));
218 logger.log(prefix + " Task time range (cycles): " + std::to_string(kTaskTimeMinCycles) + "-" +
219 std::to_string(kTaskTimeMaxCycles));
220
221 logger.log(prefix + "");
222 logger.log(prefix + "Requests");
223 logger.log(prefix + " Accepted/enqueued total: " + std::to_string(totalRequests));
224 logger.log(prefix + " Accepted during run: " + std::to_string(metrics.totalIncomingRequests));
225 logger.log(prefix + " Generated during run: " + std::to_string(totalGeneratedDuringRun) + " (accepted + rejected)");
226 logger.log(prefix + " Handled/completed: " + std::to_string(requestsHandled));
227 logger.log(prefix + " Remaining at end: " + std::to_string(requestsRemaining));
228 logger.log(prefix + " Rejected/discarded: " + std::to_string(metrics.totalRejectedRequests));
229
230 logger.log(prefix + "");
231 logger.log(prefix + "Queue");
232 logger.log(prefix + " Start -> end size: " + std::to_string(metrics.startingQueueSize) + " -> " +
233 std::to_string(endingQueueSize));
234 logger.log(prefix + " Peak size: " + std::to_string(metrics.peakQueueSize));
235 logger.log(prefix + " Average size: " + formatDouble(averageQueueSize));
236
237 logger.log(prefix + "");
238 logger.log(prefix + "Servers");
239 logger.log(prefix + " Count start -> end: " + std::to_string(initialServers) + " -> " +
240 std::to_string(balancer.getServerCount()));
241 logger.log(prefix + " Count min/max observed: " + std::to_string(metrics.minServersObserved) + " / " +
242 std::to_string(metrics.maxServersObserved));
243 logger.log(prefix + " End state active/inactive: " + std::to_string(balancer.getActiveServerCount()) + " / " +
244 std::to_string(balancer.getIdleServerCount()));
245 logger.log(prefix + " Scaling added/removed: " + std::to_string(balancer.getTotalServersAddedCount()) + " / " +
246 std::to_string(balancer.getTotalServersRemovedCount()));
247
248 logger.log(prefix + "");
249 logger.log(prefix + "Performance");
250 logger.log(prefix + " Throughput (handled/cycle): " + formatDouble(throughputPerCycle));
251 logger.log(prefix + " Completion ratio: " + formatDouble(completionRatio) + "%");
252 logger.log(prefix + " Rejection ratio: " + formatDouble(rejectionRatio) + "%");
253 logger.log(prefix + "");
254
255 logServerSnapshot(logger, balancer, prefix + "End-of-run", kServerSampleSize);
